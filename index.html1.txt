<script>

const API_URL =
"https://script.google.com/macros/s/AKfycbzoiUnYLGpxzTMC18LNFfOAWui1yFb112C-4_q0chTVKC-3F44KPXopJEsO7oFdxLR1/exec";

let semuaProduk = [];
let cart = [];

fetch(API_URL)
.then(response => response.json())
.then(data => {

  semuaProduk = data;

  tampilkanProduk(data);

})
.catch(err => {

  console.log(err);

  document.getElementById("produk")
  .innerHTML =
  "Gagal mengambil data";

});

function tampilkanProduk(data){

  let html = "";

  data.forEach(item => {

    html += `
    <div class="kartu">

      <img
      src="${item.gambar || ''}"
      >

      <h3>${item.nama}</h3>

      <p>
      Rp ${Number(item.harga)
      .toLocaleString("id-ID")}
      </p>

      <p>
      Stok: ${item.stok}
      </p>

      <button
      class="btn"
      onclick="tambahProduk(
      '${item.nama}',
      ${item.harga}
      )">
      +
      </button>

    </div>
    `;

  });

  document
  .getElementById("produk")
  .innerHTML = html;

}

function tambahProduk(nama,harga){

  const existing =
  cart.find(item =>
  item.nama === nama);

  if(existing){

    existing.qty++;

  }else{

    cart.push({
      nama:nama,
      harga:harga,
      qty:1
    });

  }

  renderCart();

}

function kurangiProduk(nama){

  const item =
  cart.find(x =>
  x.nama === nama);

  if(!item) return;

  item.qty--;

  if(item.qty <= 0){

    cart =
    cart.filter(x =>
    x.nama !== nama);

  }

  renderCart();

}

function renderCart(){

  let html = "";
  let total = 0;

  cart.forEach(item => {

    html += `
    <div class="itemKeranjang">

      <b>${item.nama}</b>

      <br><br>

      <button
      class="qtyBtn"
      onclick="kurangiProduk('${item.nama}')">
      -
      </button>

      <span style="margin:0 10px">
      ${item.qty}
      </span>

      <button
      class="qtyBtn"
      onclick="tambahProduk('${item.nama}',${item.harga})">
      +
      </button>

      <br><br>

      Rp ${(item.harga * item.qty)
      .toLocaleString("id-ID")}

    </div>
    `;

    total +=
    item.harga * item.qty;

  });

  document
  .getElementById("keranjang")
  .innerHTML = html;

  document
  .getElementById("total")
  .innerHTML =
  "Total: Rp " +
  total.toLocaleString("id-ID");

}

function bayar(){

  if(cart.length === 0){
    alert("Keranjang kosong");
    return;
  }

  fetch(API_URL,{
    method:"POST",
    headers:{
      "Content-Type":"application/json"
    },
    body:JSON.stringify({
      metode:"Tunai",
      cart:cart
    })
  })
  .then(response => response.text())
  .then(data => {

    console.log("RESPON SERVER:");
    console.log(data);

    alert(data);

  })
  .catch(error => {

    console.log("ERROR:");
    console.log(error);

    alert(error);

  });

}
document
.getElementById("cariProduk")
.addEventListener("keyup", function(){

  const keyword =
  this.value.toLowerCase();

  const hasil =
  semuaProduk.filter(item =>
    item.nama
    .toLowerCase()
    .includes(keyword)
  );

  tampilkanProduk(hasil);

});
